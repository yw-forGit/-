Flume最主要的作用就是，实时读取服务器本地磁盘的数据，将数据写入到HDFS。

编辑flume-telnet-logger.cof

# >> a1是agent的名称。r1、k1、c1是agent的三个组件

# r1是a1的输入源

a1.sources = r1

# k1是a1的输出目的地

a1.sinks = k1

# c1是a1的缓冲区

a1.channels = c1

 

# >> source配置。本例子，使用telnet往localhost:44444发送数据

# a1的输入类型为netcat端口类型

a1.sources.r1.type = netcat

# a1的监听的主机

a1.sources.r1.bind = localhost

# a1的监听的端口号

a1.sources.r1.port = 44444

 

# >> sink配置

# a1的输出目的地是控制台logger

a1.sinks.k1.type = logger

 

# >> channel的内存

# a1的channel类型是memory内存型

a1.channels.c1.type = memory

# a1的channel总容量1000个event

a1.channels.c1.capacity = 1000

# a1的channel传输时收集到了100条event以后再去提交事务

a1.channels.c1.transactionCapacity = 100

 

# >> 拓扑结构

# put过程。将r1与c1连接起来

a1.sources.r1.channels = c1

# take过程。将k1与c1连接起来

a1.sinks.k1.channel = c1

编辑flume-file-hdfs.conf

# >> Agent

a2.sources = r2

a2.sinks = k2

a2.channels = c2

 

# >> Source

# source类型为exec可执行命令

a2.sources.r2.type = exec

# 具体执行的命令。监听hive.log文件内容

a2.sources.r2.command = tail -F /usr/local/flume/hive_temp/hive.log

# 执行shell脚本的绝对路径

a2.sources.r2.shell = /bin/bash -c

 

# >> Sink

a2.sinks.k2.type = hdfs

# 写入到HDFS。按照时间滚动文件夹的格式。%Y是四位年份，%y是两位年份

a2.sinks.k2.hdfs.path = hdfs://master:9000/flume/%y-%m-%d/%H

# 上传文件的前缀

a2.sinks.k2.hdfs.filePrefix = logs-

# 是否按照时间滚动文件夹

a2.sinks.k2.hdfs.round = true

# 多少时间单位创建一个新的文件夹

a2.sinks.k2.hdfs.roundValue = 1

# 重新定义时间单位为小时

a2.sinks.k2.hdfs.roundUnit = hour

# 是否使用本地时间戳

a2.sinks.k2.hdfs.useLocalTimeStamp = true

# 积攒多少个Event才flush到HDFS一次

a2.sinks.k2.hdfs.batchSize = 1000

# 设置文件类型，可支持压缩

a2.sinks.k2.hdfs.fileType =  DataStream

# 多久生成一个新的文件，600=10分钟

a2.sinks.k2.hdfs.rollInterval = 600

# 设置每个文件的滚动大小

a2.sinks.k2.rollSize = 134217700

# 文件的滚动与Event数量无关

a2.sinks.k2.hdfs.rollCount = 0

# 最小冗余数

a2.sinks.k2.hdfs.minBlockReplicas = 1

 

# >> channel的内存

# a2的channel类型是memory内存型

a2.channels.c2.type = memory

# a2的channel总容量1000个event

a2.channels.c2.capacity = 1000

# a2的channel传输时收集到了100条event以后再去提交事务

a2.channels.c2.transactionCapacity = 100

 

 

# >> 拓扑结构

# put过程。将r2与c2连接起来

a2.sources.r2.channels = c2

# take过程。将k2与c2连接起来

a2.sinks.k2.channel = c2

单数据源多出口案例1

原型

单Source，多Channel，多Sink

 

 

分析

需求：使用Flume-1监控文件变动，Flume-1将变动内容传递给Flume-2，Flume-2负责存储HDFS。同时Flume-1将变动内容传递给Flume-3，Flume-3负责输出到本地文件系统。（Flume-2、Flume-3获取到的都是一样的数据）

 

Flume与Flume之间是Avro，Avro是一种与语言无关的数据序列化和RPC框架。

在flume/job/group1下创建

 

看官方文档

Avro Sink是使用hostname，Avro Source是bind（看文档五分钟，省去debug两小时），bind是服务器

# >> Agent

a1.sources = r1

a1.sinks = k1 k2

a1.channels = c1 c2

# 将数据流复制给所有channel

a1.sources.r1.selector.type = replicating

a1.sinks.k1.hostname = slave1

a1.sinks.k2.type = avro

进入group3，在slave1上创建文件flume-logger-flume.conf

 

编辑

# Agent

a1.sources = r1

a1.sinks = k1

a1.channels = c1

 

# Source

a1.sources.r1.type = exec

a1.sources.r1.command = tail -F /usr/local/flume/group3.log

a1.sources.r1.shell = /bin/bash -c

 

# Sink

a1.sinks.k1.type = avro

# 聚合到slave3的Flume

a1.sinks.k1.hostname = slave3

a1.sinks.k1.port = 4141

 

# Channel

a1.channels.c1.type = memory

a1.channels.c1.capacity = 1000

a1.channels.c1.transactionCapacity = 100

 

# 拓扑结构

a1.sources.r1.channels = c1

a1.sinks.k1.channel = c1

 

flume-netcat-flume.conf

进入group3，在slave2上创建文件flume-netcat-flume.conf

 

编辑

# Agent

a2.sources = r1

a2.sinks = k1

a2.channels = c1

 

# Source

a2.sources.r1.type = netcat

a2.sources.r1.bind = localhost

a2.sources.r1.port = 44444

 

# Sink

a2.sinks.k1.type = avro

# 聚合到slave3的Flume

a2.sinks.k1.hostname = slave3

a2.sinks.k1.port = 4141

 

# Channel

a2.channels.c1.type = memory

a2.channels.c1.capacity = 1000

a2.channels.c1.transactionCapacity = 100

 

# 拓扑结构

a2.sources.r1.channels = c1

a2.sinks.k1.channel = c1

 

 

flume-flume-logger.conf

进入group3，在slave3上创建文件flume-flume-logger.conf

 

# Agent

a3.sources = r1

a3.sinks = k1

a3.channels = c1

 

# Source

a3.sources.r1.type = avro

# 前两个Flume都聚合到这里了

a3.sources.r1.bind = slave3

a3.sources.r1.port = 4141

 

# Sink

a3.sinks.k1.type = logger

 

# Channel

a3.channels.c1.type = memory

a3.channels.c1.capacity = 1000

a3.channels.c1.transactionCapacity = 100

 

# 拓扑结构

a3.sources.r1.channels = c1

a3.sinks.k1.channel = c1

 

 

启动Agent

先a3

bin/flume-ng agent --conf conf/ --name a3 --conf-file job/group3/flume-flume-logger.conf -Dflume.root.logger=INFO,console

 

a2 可以不加日志

bin/flume-ng agent --conf conf/ --name a2 --conf-file job/group3/flume-netcat-flume.conf -Dflume.root.logger=INFO,console

 

a1

bin/flume-ng agent --conf conf/ --name a1 --conf-file job/group3/flume-logger-flume.conf -Dflume.root.logger=INFO,console
